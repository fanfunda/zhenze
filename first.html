git checkout 编号 表示回到这个编号更改

git checkout - 表示回到最新一次更改

git log --all --graph 表示图形化所有分支

git tag -a 标签名 -m "备注" 表示添加标签

git show 标签名 -m 表示查看标签详细信息

git touch 文件 表示在当前目录下创建文件

git remote add 远程名称 远程地址 表示添加远程仓库

git push -u 远程名 分支名 表示上传指定分支到远程仓库

git config --global user.name "name" 全局设置提交者姓名

git config --global user.email "email" 全局设置提交者邮箱

git merge 分支名 将分支与当前分支合并

git push -u 远程名 分支名 表示下载指定分支到远程仓库

git remote rm 别名 删除远程仓库
